```javascript
const fs = require('fs');
const path = require('path');
const STASH_PATH = path.join(__dirname, '..', 'memory', 'invention-training', 'pollen-stash.json');

function load() {
  try { return JSON.parse(fs.readFileSync(STASH_PATH, 'utf8')); }
  catch { return { items: [], created: new Date().toISOString() }; }
}

function save(data) {
  fs.writeFileSync(STASH_PATH, JSON.stringify(data, null, 2));
}

function add(content, energy, domain, ttlHours) {
  const stash = load();
  const item = {
    id: Date.now().toString(36),
    content,
    energy: parseInt(energy) || 3,
    domain: domain || 'general',
    created: new Date().toISOString(),
    expiresAt: new Date(Date.now() + (parseInt(ttlHours) || 48) * 3600000).toISOString(),
    used: false
  };
  stash.items.push(item);
  // Prune expired
  stash.items = stash.items.filter(i => new Date(i.expiresAt) > new Date());
  save(stash);
  console.log(JSON.stringify({ action: 'added', item }));
}

function pop(count) {
  const stash = load();
  const available = stash.items.filter(i => !i.used);
  available.sort((a, b) => b.energy - a.energy);
  const top = available.slice(0, parseInt(count) || 3);
  top.forEach(i => { i.used = true; i.usedAt = new Date().toISOString(); });
  save(stash);
  console.log(JSON.stringify({ action: 'popped', items: top }));
}

function list() {
  const stash = load();
  const active = stash.items.filter(i => !i.used && new Date(i.expiresAt) > new Date());
  console.log(JSON.stringify({ active: active.length, items: active }));
}

const [,, cmd, ...args] = process.argv;
if (cmd === 'add') add(args[0], args[1]?.split('=')[1], args[2]?.split('=')[1], args[3]?.split('=')[1]);
else if (cmd === 'pop') pop(args[0]?.split('=')[1]);
else if (cmd === 'list') list();
else console.log('Usage: add <content> --energy N --domain X --ttl H | pop --count N | list');
```
